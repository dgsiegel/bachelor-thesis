% Commands to be used within the TUM report document

% comment that appears on the border
\newcommand{\comment}[1]{\marginpar{\raggedright \noindent
\textcolor{scarletred1}{\textbf{FIX ME:}} \footnotesize {\sl #1} }}

% page clearing
\newcommand{\clearemptydoublepage}{%
  \ifthenelse{\boolean{@twoside}}{\newpage{\pagestyle{empty}\cleardoublepage}}%
  {\clearpage}}

% more space between rows of a table
\newcommand{\ra}[1]{\renewcommand{\arraystretch}{#1}}

\let\oldquote\quote
\renewcommand\quote{\par\small\oldquote}

% custom source code inclusion
\newcommand{\lstinputlistingln}[3]{\vspace{1em}\lstinputlisting[firstnumber=#1,linerange=#1-#2]{#3}\vspace{1em}}
